<script>
    const tempS = document.getElementById('temp-slider');
    const visS = document.getElementById('vis-slider');
    const fluxS = document.getElementById('weather-flux');
    const statusD = document.getElementById('status-display');

    function analyzeVariables() {
        const temp = parseInt(tempS.value);
        const vis = parseInt(visS.value);
        const flux = fluxS.value;

        document.getElementById('temp-display').innerText = temp;
        document.getElementById('vis-display').innerText = vis;

        let riskScore = 0;

        // Biological Window: 14-24°C is high metabolism/hunting window
        if (temp >= 14 && temp <= 24) riskScore += 2;
        
        // Low visibility increases ambush potential
        if (vis < 4) riskScore += 2;
        
        // Rainfall runoff increases organic matter/prey presence
        if (flux === 'runoff') riskScore += 2;

        if (riskScore >= 4) {
            statusD.innerText = "CRITICAL: ENVIRONMENTAL ANOMALY DETECTED";
            statusD.className = "status-box risk-critical";
        } else if (riskScore >= 2) {
            statusD.innerText = "ELEVATED: PREDATORY WINDOW OPEN";
            statusD.className = "status-box risk-elevated";
        } else {
            statusD.innerText = "STABLE: LOW CORRELATION RISK";
            statusD.className = "status-box risk-low";
        }
    }

    [tempS, visS, fluxS].forEach(el => el.addEventListener('input', analyzeVariables));
    analyzeVariables();
</script>
